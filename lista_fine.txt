    <script src="https://cdnjs.cloudflare.com/ajax/libs/babel-core/5.8.24/browser.js"></script>
    <script type="text/babel">
      const {Component} = React;
      class App extends Component {
        constructor(props) {
          super(props)
          this.state = {
            lista: [],
            inserito:''
          }
        }

        aggiungi(evento){
          let lis = this.state.lista;
          lis.push(<li>{this.state.inserito}</li>)
          this.setState({
            lista: lis
          });
        }

        insAttivita(event){
          this.setState({
            inserito:event.target.value
          });
        }

        render(){
          return (
            <div>
            <h1>hai {this.state.lista.length} attivit√† da svolgere</h1>
            <ul>
            {this.state.lista}
            </ul>
            <input onChange={this.insAttivita.bind(this)} />
            <button onClick={this.aggiungi.bind(this)}>Aggiungi</button>
            </div>
          )
        }
      }

      ReactDOM.render(<App/>, document.getElementById('app'))



    </script>
